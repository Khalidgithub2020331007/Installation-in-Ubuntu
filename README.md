# Apache
    sudo apt update
    sudo apt install apache2
    sudo ufw app list
    sudo ufw allow 'Apache'
    sudo ufw status
    sudo systemctl status apache2

<br><br>

# Unistall Apache
    sudo systemctl stop apache2
    sudo apt purge apache2
    sudo apt autoremove
